rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    /* ==============================
       HELPER FUNCTIONS
    ============================== */

    function isAuthenticated() {
      return request.auth != null;
    }

    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }

    function isParticipant(participants) {
      return isAuthenticated() && participants.contains(request.auth.uid);
    }

    function isValidRole(role) {
      return role in ['owner', 'tenant'];
    }

    function isValidPropertyStatus(status) {
      return status in ['approved', 'pending', 'booked', 'rented', 'ACTIVE', 'active', 'available', 'BOOKED'];
    }

    function isValidPropertyType(type) {
      return type in ['apartment', 'house', 'condo', 'studio', 'villa', 'pg', 'flat', 'commercial'];
    }

    function isValidBookingStatus(status) {
      return status in ['pending', 'pending_payment', 'paid', 'confirmed', 'active', 'cancelled', 'completed'];
    }

    function propertyExists(propertyId) {
      return exists(/databases/$(database)/documents/properties/$(propertyId));
    }

    function getProperty(propertyId) {
      return get(/databases/$(database)/documents/properties/$(propertyId));
    }

    function userExists(userId) {
      return exists(/databases/$(database)/documents/users/$(userId));
    }

    /* ==============================
       USERS COLLECTION (UPDATED FOR BANK DETAILS)
    ============================== */

    match /users/{userId} {
      allow read: if isOwner(userId);

      allow create: if isAuthenticated() &&
                       isOwner(userId) &&
                       request.resource.data.keys().hasOnly(['name','email','role','createdAt']) &&
                       request.resource.data.name is string &&
                       request.resource.data.email is string &&
                       request.resource.data.createdAt is timestamp &&
                       isValidRole(request.resource.data.role);

      allow update: if isOwner(userId) &&
                       (request.resource.data.diff(resource.data).affectedKeys().hasAny(['name','email','bankDetails']) ||
                        request.resource.data.diff(resource.data).affectedKeys().size == 0) &&
                       (!request.resource.data.diff(resource.data).affectedKeys().hasAll(['role']) || resource.data.role == request.resource.data.role) &&
                       (!request.resource.data.diff(resource.data).affectedKeys().hasAll(['createdAt']) || resource.data.createdAt == request.resource.data.createdAt) &&
                       (request.resource.data.name is string || request.resource.data.name == null) &&
                       (request.resource.data.email is string || request.resource.data.email == null);

      allow delete: if isOwner(userId);
    }

    /* ==============================
       PROPERTIES COLLECTION
    ============================== */

    match /properties/{propertyId} {
      allow read: if resource.data.status == 'approved' ||
                     (isAuthenticated() && resource.data.owner_uid == request.auth.uid);

      allow create: if isAuthenticated() &&
                       request.resource.data.owner_uid == request.auth.uid &&
                       request.resource.data.keys().hasOnly([
                         'owner_uid','title','description','price',
                         'location','type','images','status','createdAt'
                       ]) &&
                       request.resource.data.title is string &&
                       request.resource.data.description is string &&
                       request.resource.data.price is number &&
                       request.resource.data.location is string &&
                       request.resource.data.createdAt is timestamp &&
                       isValidPropertyType(request.resource.data.type) &&
                       isValidPropertyStatus(request.resource.data.status);

      allow update: if isAuthenticated() &&
                       resource.data.owner_uid == request.auth.uid &&
                       request.resource.data.keys().hasOnly([
                         'title','description','price','location',
                         'type','images','status'
                       ]) &&
                       (!('type' in request.resource.data) ||
                        isValidPropertyType(request.resource.data.type)) &&
                       (!('status' in request.resource.data) ||
                        isValidPropertyStatus(request.resource.data.status));

      allow delete: if isAuthenticated() &&
                       resource.data.owner_uid == request.auth.uid;
    }

    /* ==============================
       BOOKINGS COLLECTION
    ============================== */

    match /bookings/{bookingId} {
      allow read: if isAuthenticated() &&
                     (resource.data.tenantId == request.auth.uid ||
                      resource.data.ownerId == request.auth.uid);

      allow create: if isAuthenticated() &&
                       request.resource.data.tenantId == request.auth.uid &&
                       request.resource.data.keys().hasOnly([
                         'propertyId','tenantId','ownerId',
                         'status','createdAt'
                       ]) &&
                       request.resource.data.createdAt is timestamp &&
                       isValidBookingStatus(request.resource.data.status) &&
                       propertyExists(request.resource.data.propertyId) &&
                       getProperty(request.resource.data.propertyId).data.owner_uid
                         == request.resource.data.ownerId &&
                       getProperty(request.resource.data.propertyId).data.status
                         == 'approved';

      allow update: if isAuthenticated() &&
                       resource.data.ownerId == request.auth.uid &&
                       request.resource.data.keys().hasOnly(['status']) &&
                       isValidBookingStatus(request.resource.data.status);

      allow delete: if isAuthenticated() &&
                       resource.data.ownerId == request.auth.uid;
    }

    /* ==============================
       CHATS COLLECTION
    ============================== */

    match /chats/{chatId} {
      allow read: if isAuthenticated() &&
               (resource.data.tenantId == request.auth.uid ||
                resource.data.ownerId == request.auth.uid);
  
      allow create: if isAuthenticated() &&
               (request.resource.data.tenantId == request.auth.uid ||
                request.data.ownerId == request.auth.uid) &&
               request.resource.data.participants is list &&
               request.resource.data.participants.hasAll([request.auth.uid]);
  
      allow update: if isAuthenticated() &&
               (resource.data.tenantId == request.auth.uid ||
                request.data.ownerId == request.auth.uid);
  
      allow delete: if isAuthenticated() &&
               (resource.data.tenantId == request.auth.uid ||
                request.data.ownerId == request.auth.uid);

      // ── MESSAGES SUBCOLLECTION ─────────────────────────────────────
      match /messages/{messageId} {
        allow read: if isAuthenticated() &&
               get(/databases/$(database)/documents/chats/$(chatId)).data.tenantId == request.auth.uid ||
               get(/databases/$(database)/documents/chats/$(chatId)).data.ownerId == request.auth.uid;
    
        allow create: if isAuthenticated() &&
                 request.resource.data.senderId == request.auth.uid;
    
        allow update: if false;
        allow delete: if isAuthenticated() &&
                 resource.data.senderId == request.auth.uid;
      }
    }

    /* ==============================
       NOTIFICATIONS COLLECTION
    ============================== */

    match /notifications/{notifId} {
      allow read: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow create: if isAuthenticated();
      allow update: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }

    /* ==============================
       WISHLISTS COLLECTION
    ============================== */

    match /wishlists/{wishId} {
      allow read: if isAuthenticated() && resource.data.tenantId == request.auth.uid;
      allow create: if isAuthenticated() &&
                       request.resource.data.tenantId == request.auth.uid &&
                       request.resource.data.keys().hasOnly(['tenantId','propertyId','createdAt']);
      allow delete: if isAuthenticated() && request.data.tenantId == request.auth.uid;
    }

    /* ==============================
       AGREEMENTS COLLECTION
    ============================== */

    match /agreements/{agreementId} {
      allow read: if isAuthenticated() &&
                     (resource.data.tenantId == request.auth.uid ||
                      resource.data.ownerId == request.auth.uid);
      allow create: if isAuthenticated();
      allow update: if isAuthenticated() &&
                       (resource.data.tenantId == request.auth.uid ||
                        resource.data.ownerId == request.auth.uid);
    }

    /* ==============================
       TENANT DETAILS COLLECTION
    ============================== */

    match /tenantDetails/{tenantId} {
      allow read, write: if isAuthenticated() && request.auth.uid == tenantId;
    }

    /* ==============================
       TENANT DOCUMENTS COLLECTION
    ============================== */

    match /tenant_documents/{userId} {
      allow read, write: if isAuthenticated() && request.auth.uid == userId;
    }

    /* ==============================
       ACTIVITY LOGS COLLECTION
    ============================== */

    match /activity_logs/{logId} {
      allow read: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow create: if isAuthenticated();
    }

    /* ==============================
       DENY ALL OTHER ACCESS
    ============================== */

    match /{document=**} {
      allow read, write: if false;
    }
  }
}

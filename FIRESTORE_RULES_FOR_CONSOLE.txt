rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper functions
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }
    
    function isParticipant(participants) {
      return isAuthenticated() && participants.contains(request.auth.uid);
    }
    
    function isValidRole(role) {
      return role in ['owner', 'tenant'];
    }
    
    function isValidBookingStatus(status) {
      return status in ['pending', 'confirmed', 'cancelled'];
    }
    
    function isValidPropertyStatus(status) {
      return status in ['approved', 'pending'];
    }
    
    function isValidPropertyType(type) {
      return type in ['apartment', 'house', 'condo', 'studio', 'villa'];
    }

    // USERS COLLECTION
    match /users/{userId} {
      // Read: Only own user document
      allow read: if isOwner(userId);
      
      // Create: Only authenticated users can create their own document
      allow create: if isAuthenticated() && 
                       isOwner(userId) &&
                       request.resource.data.keys().hasAll(['name', 'email', 'role']) &&
                       request.resource.data.keys().hasOnly(['name', 'email', 'role', 'createdAt']) &&
                       isValidRole(request.resource.data.role) &&
                       request.resource.data.name is string &&
                       request.resource.data.email is string &&
                       request.resource.data.createdAt is timestamp;
      
      // Update: Only own document, but cannot change role
      allow update: if isOwner(userId) &&
                       request.resource.data.keys().hasOnly(['name', 'email']) &&
                       (!request.resource.data.keys().hasAll(['role']) || resource.data.role == request.resource.data.role) &&
                       (request.resource.data.name is string || request.resource.data.name == null) &&
                       (request.resource.data.email is string || request.resource.data.email == null);
      
      // Delete: Only own document
      allow delete: if isOwner(userId);
    }

    // PROPERTIES COLLECTION
    match /properties/{propertyId} {
      // Read: Anyone can read approved properties, owners can read their own properties regardless of status
      allow read: if (resource.data.status == 'approved') ||
                     (isAuthenticated() && resource.data.ownerId == request.auth.uid);
      
      // Create: Only owners can create properties and must be the owner
      allow create: if isAuthenticated() &&
                       request.resource.data.ownerId == request.auth.uid &&
                       request.resource.data.keys().hasAll(['ownerId', 'title', 'description', 'price', 'location', 'type', 'status', 'createdAt']) &&
                       request.resource.data.keys().hasOnly(['ownerId', 'title', 'description', 'price', 'location', 'type', 'images', 'status', 'createdAt']) &&
                       isValidPropertyType(request.resource.data.type) &&
                       isValidPropertyStatus(request.resource.data.status) &&
                       request.resource.data.title is string &&
                       request.resource.data.description is string &&
                       request.resource.data.price is number &&
                       request.resource.data.location is string &&
                       (request.resource.data.images is list || request.resource.data.images == null) &&
                       request.resource.data.createdAt is timestamp;
      
      // Update: Only owners can update their own properties
      allow update: if isAuthenticated() &&
                       resource.data.ownerId == request.auth.uid &&
                       request.resource.data.keys().hasOnly(['title', 'description', 'price', 'location', 'type', 'images', 'status']) &&
                       (!request.resource.data.keys().hasAll(['type']) || isValidPropertyType(request.resource.data.type)) &&
                       (!request.resource.data.keys().hasAll(['status']) || isValidPropertyStatus(request.resource.data.status)) &&
                       (request.resource.data.title is string || request.resource.data.title == null) &&
                       (request.resource.data.description is string || request.resource.data.description == null) &&
                       (request.resource.data.price is number || request.resource.data.price == null) &&
                       (request.resource.data.location is string || request.resource.data.location == null) &&
                       (request.resource.data.images is list || request.resource.data.images == null);
      
      // Delete: Only owners can delete their own properties
      allow delete: if isAuthenticated() && resource.data.ownerId == request.auth.uid;
    }

    // BOOKINGS COLLECTION
    match /bookings/{bookingId} {
      // Read: Tenants can read their own bookings, owners can read bookings for their properties
      allow read: if isAuthenticated() && 
                     (resource.data.tenantId == request.auth.uid || resource.data.ownerId == request.auth.uid);
      
      // Create: Only tenants can create bookings for themselves
      allow create: if isAuthenticated() &&
                       request.resource.data.tenantId == request.auth.uid &&
                       request.resource.data.keys().hasAll(['propertyId', 'tenantId', 'ownerId', 'status', 'createdAt']) &&
                       request.resource.data.keys().hasOnly(['propertyId', 'tenantId', 'ownerId', 'status', 'createdAt']) &&
                       isValidBookingStatus(request.resource.data.status) &&
                       request.resource.data.propertyId is string &&
                       request.resource.data.ownerId is string &&
                       request.resource.data.createdAt is timestamp;
      
      // Update: Only owners can update booking status
      allow update: if isAuthenticated() &&
                       resource.data.ownerId == request.auth.uid &&
                       request.resource.data.keys().hasOnly(['status']) &&
                       isValidBookingStatus(request.resource.data.status);
      
      // Delete: Only owners can delete bookings
      allow delete: if isAuthenticated() && resource.data.ownerId == request.auth.uid;
    }

    // CHATS COLLECTION
    match /chats/{chatId} {
      // Read: Only participants can read chat
      allow read: if isParticipant(resource.data.participants);
      
      // Create: Only authenticated users can create chats and must be a participant
      allow create: if isAuthenticated() &&
                       request.resource.data.participants.contains(request.auth.uid) &&
                       request.resource.data.keys().hasAll(['participants', 'updatedAt']) &&
                       request.resource.data.keys().hasOnly(['participants', 'lastMessage', 'updatedAt']) &&
                       request.resource.data.participants is list &&
                       request.resource.data.participants.size() == 2 &&
                       request.resource.data.updatedAt is timestamp &&
                       (request.resource.data.lastMessage is string || request.resource.data.lastMessage == null);
      
      // Update: Only participants can update chat (lastMessage, updatedAt)
      allow update: if isParticipant(resource.data.participants) &&
                       request.resource.data.keys().hasOnly(['lastMessage', 'updatedAt']) &&
                       (request.resource.data.lastMessage is string || request.resource.data.lastMessage == null) &&
                       request.resource.data.updatedAt is timestamp;
      
      // Delete: Only participants can delete chat
      allow delete: if isParticipant(resource.data.participants);
    }

    // MESSAGES COLLECTION
    match /messages/{messageId} {
      // Read: Only chat participants can read messages
      allow read: if isAuthenticated() && 
                     exists(/databases/$(database)/documents/chats/$(resource.data.chatId)) &&
                     isParticipant(get(/databases/$(database)/documents/chats/$(resource.data.chatId)).data.participants);
      
      // Create: Only authenticated users can create messages and must be the sender
      allow create: if isAuthenticated() &&
                       request.resource.data.senderId == request.auth.uid &&
                       request.resource.data.keys().hasAll(['chatId', 'senderId', 'text', 'createdAt']) &&
                       request.resource.data.keys().hasOnly(['chatId', 'senderId', 'text', 'createdAt']) &&
                       request.resource.data.chatId is string &&
                       request.resource.data.text is string &&
                       request.resource.data.createdAt is timestamp &&
                       // Verify sender is a participant in the chat
                       exists(/databases/$(database)/documents/chats/$(request.resource.data.chatId)) &&
                       get(/databases/$(database)/documents/chats/$(request.resource.data.chatId)).data.participants.contains(request.auth.uid);
      
      // Update: Messages are immutable - no updates allowed
      allow update: if false;
      
      // Delete: Only sender can delete their own messages
      allow delete: if isAuthenticated() && resource.data.senderId == request.auth.uid;
    }

    // DENY ALL OTHER ACCESS
    match /{document=**} {
      allow read, write: if false;
    }
  }
}

service firebase.storage {
  match /b/{bucket}/o {
    
    // Helper functions
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }

    // PROPERTY IMAGES STORAGE
    // Structure: propertyImages/{ownerId}/{propertyId}/{imageName}
    match /propertyImages/{ownerId}/{propertyId}/{allPaths=**} {
      // Read: Anyone can read property images (for approved properties)
      allow read: if true;
      
      // Write: Only property owners can upload/update/delete their property images
      allow write: if isOwner(ownerId);
    }

    // USER DOCUMENTS STORAGE
    // Structure: userDocuments/{userId}/{fileName}
    match /userDocuments/{userId}/{allPaths=**} {
      // Read: Only users can read their own documents
      allow read: if isOwner(userId);
      
      // Write: Only users can upload/update/delete their own documents
      allow write: if isOwner(userId);
    }

    // DENY ALL OTHER ACCESS
    match /{path=**} {
      allow read, write: if false;
    }
  }
}

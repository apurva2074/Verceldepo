service firebase.storage {
  match /b/{bucket}/o {

    function isAuthenticated() {
      return request.auth != null;
    }

    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }

    function propertyExists(propertyId) {
      return exists(/databases/(default)/documents/properties/$(propertyId));
    }

    function propertyBelongsToOwner(propertyId, ownerId) {
      return propertyExists(propertyId) &&
             get(/databases/(default)/documents/properties/$(propertyId)).data.ownerId == ownerId;
    }

    /* PROPERTY IMAGES */
    match /propertyImages/{ownerId}/{propertyId}/{allPaths=**} {

      allow read: if propertyExists(propertyId) &&
                     get(/databases/(default)/documents/properties/$(propertyId)).data.status == 'approved';

      allow write: if isOwner(ownerId) &&
                      propertyBelongsToOwner(propertyId, ownerId);
    }

    /* USER DOCUMENTS */
    match /userDocuments/{userId}/{allPaths=**} {

      allow read, write: if isOwner(userId);
    }

    match /{path=**} {
      allow read, write: if false;
    }
  }
}
